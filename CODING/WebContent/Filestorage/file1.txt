package com.sw02control;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.io.PrintWriter;
import java.util.Date;

import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.servlet.http.HttpSession;

import com.sw02bean.EmpBean;
import com.sw02dao.Sw02Dao;

/**
 * Servlet implementation class Working
 */
@WebServlet("/Working")
public class Working extends HttpServlet {
	private static final long serialVersionUID = 1L;
       
    /**
     * @see HttpServlet#HttpServlet()
     */
    public Working() {
        super();
        // TODO Auto-generated constructor stub
    }

	/**
	 * @see HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		// TODO Auto-generated method stub
	}

	/**
	 * @see HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		// TODO Auto-generated method stub
		PrintWriter o = response.getWriter();
		HttpSession session = request.getSession(false);
		String uid = (String) session.getAttribute("email");
		String fname = request.getParameter("fname");
		String sub = request.getParameter("sub");
		sub = "Subject: " + sub + "\n";
		String data = request.getParameter("data");
		fname = "D:\\VTJSW02\\" + fname;
		int count = Integer.parseInt(request.getParameter("count"));
		int kcount = Integer.parseInt(request.getParameter("kcount"));
		File f = new File(fname);
		FileOutputStream fo = new FileOutputStream(f);
		if( !f.exists()){
			f.createNewFile();
		}
		byte b [] = data.getBytes();
		fo.write(b);
		fo.flush();
		fo.close();
		InputStream photo = new FileInputStream(f);
		Date d = new Date();
		int mo = d.getMonth() + 1;
		String dt = "" + d.getDate() + "/" + mo;
		String s = "" + d;
		String sql = "insert into empreport1 values(?,?,?,?,?,?)";
		EmpBean eb = new EmpBean();
		eb.setUid(uid);
		eb.setCn(fname);
		eb.setDa(s);
		eb.setSub(sub);
		eb.setPhoto(photo);
		eb.setContent("application/octet-stream");
		int i = Sw02Dao.empReport(sql, eb);
		eb.setCount(count);
		eb.setKcount(kcount);
		eb.setCn("Working.java");
		eb.setDa(dt);
		sql = "insert into empclick values(?,?,?,?,?)";
		int j = Sw02Dao.clicks(sql, eb);
		if(j > 0 && i >0){
			o.println("<script type=\"text/javascript\">");
			o.println("alert('File Stored Successfully...');");
			o.println("window.location='work1.jsp';</script>");
		}else{
			o.println("<script type=\"text/javascript\">");
			o.println("alert('File not stored...');");
			o.println("window.location='projectwork.jsp';</script>");
		}
	}

}
